#version 400 core

in vec2 position;

out vec4 clipSpace;
out vec2 textureCoords;

uniform mat4 projectionMatrix, viewMatrix, modelMatrix;

// Valor para colocar en mosaicos las coordendas de textura
const float tiling = 6.0;

void main(void) {

    // Genera las coordenadas del espacio de recorte de este vertice
    clipSpace =  projectionMatrix * viewMatrix * modelMatrix * vec4(position.x, 0.0, position.y, 1.0);
	gl_Position = clipSpace;
	textureCoords = vec2(position.x / 2.0 + 0.5, position.y / 2.0 + 0.5) * tiling;
 
}
